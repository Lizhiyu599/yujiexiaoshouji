  <script type="module">
    // ==========================================
    // 玉界机 · 桌面系统底层框架
    // 模块化架构：页面管理、网格系统、Dock系统、元素管理
    // 当前阶段仅搭建架构，无具体应用，无写死内容
    // ==========================================

    // ---------- 核心数据结构 ----------
    /**
     * 桌面元素统一结构
     * @typedef {Object} DesktopItem
     * @property {string} id - 唯一ID
     * @property {'app'|'widget'} type - 元素类型
     * @property {number} pageIndex - 所在页面索引(从0开始)
     * @property {number} gridX - 网格列起始坐标 (0-5)
     * @property {number} gridY - 网格行起始坐标 (0-3)
     * @property {number} width - 占位宽度(列数)
     * @property {number} height - 占位高度(行数)
     * @property {string} name - 显示名称
     * @property {Object} [meta] - 扩展信息
     */

    // 全局状态：桌面元素注册表 (当前为空，未来通过管理接口添加)
    const desktopItems = new Map(); // key: id, value: DesktopItem

    // 页面管理状态
    const pageState = {
      count: 2,             // 动态页数，初始2页用于演示分页架构
      currentPage: 0,
    };

    // Dock 数据 (最多4个槽位，结构预留)
    const dockState = {
      items: [],            // 存放应用ID或null，长度固定4
      maxSlots: 4,
    };

    // ---------- 模块：桌面元素管理器 ----------
    const ElementManager = {
      /**
       * 注册元素 (未来添加图标/小组件时调用)
       * @param {DesktopItem} item
       */
      addItem(item) {
        if (!item.id || !item.type) return;
        desktopItems.set(item.id, { ...item });
        this.renderAllPages();
      },

      removeItem(id) {
        desktopItems.delete(id);
        this.renderAllPages();
      },

      getItemsByPage(pageIndex) {
        return Array.from(desktopItems.values()).filter(item => item.pageIndex === pageIndex);
      },

      /**
       * 根据网格坐标生成CSS grid-area
       */
      getGridStyle(item) {
        const colStart = item.gridX + 1;
        const colEnd = colStart + item.width;
        const rowStart = item.gridY + 1;
        const rowEnd = rowStart + item.height;
        return {
          gridColumn: `${colStart} / ${colEnd}`,
          gridRow: `${rowStart} / ${rowEnd}`,
        };
      },

      renderAllPages() {
        // 触发页面重新渲染
        PageManager.renderPages();
      }
    };

    // ---------- 模块：网格系统 (独立管理网格逻辑) ----------
    const GridSystem = {
      columns: 6,
      rows: 4,

      /**
       * 验证占位是否在网格内
       */
      isValidPlacement(gridX, gridY, width, height) {
        return (
          gridX >= 0 && gridY >= 0 &&
          gridX + width <= this.columns &&
          gridY + height <= this.rows
        );
      },

      /**
       * 检测占位冲突 (未来编辑模式使用)
       */
      checkCollision(pageIndex, ignoreId, gridX, gridY, width, height) {
        const items = ElementManager.getItemsByPage(pageIndex);
        return items.some(item => {
          if (item.id === ignoreId) return false;
          const xOverlap = gridX < item.gridX + item.width && gridX + width > item.gridX;
          const yOverlap = gridY < item.gridY + item.height && gridY + height > item.gridY;
          return xOverlap && yOverlap;
        });
      }
    };

    // ---------- 模块：Dock 系统 ----------
    const DockSystem = {
      render() {
        const dockContainer = document.getElementById('dockItems');
        if (!dockContainer) return;
        dockContainer.innerHTML = '';

        // 固定4个槽位
        for (let i = 0; i < this.maxSlots; i++) {
          const slot = document.createElement('div');
          slot.className = 'dock-slot';
          slot.dataset.slotIndex = i;
          
          const itemId = dockState.items[i] || null;
          if (itemId && desktopItems.has(itemId)) {
            const item = desktopItems.get(itemId);
            slot.textContent = item.name?.charAt(0) || '?';
            slot.style.color = 'white';
            slot.style.background = 'rgba(255,255,255,0.15)';
          } else {
            slot.textContent = '';
            slot.style.background = 'rgba(255,255,255,0.06)';
          }
          
          // 预留拖拽事件 (本阶段仅结构)
          dockContainer.appendChild(slot);
        }
      },

      get maxSlots() {
        return dockState.maxSlots;
      },

      /**
       * 未来添加应用到Dock
       */
      addToDock(itemId) {
        if (dockState.items.length >= this.maxSlots) return false;
        if (!dockState.items.includes(itemId)) {
          dockState.items.push(itemId);
          this.render();
          return true;
        }
        return false;
      },

      removeFromDock(itemId) {
        const index = dockState.items.indexOf(itemId);
        if (index > -1) {
          dockState.items.splice(index, 1);
          this.render();
        }
      }
    };

    // ---------- 模块：页面管理 & 分页系统 ----------
    const PageManager = {
      viewport: null,
      indicator: null,

      init() {
        this.viewport = document.getElementById('pagesViewport');
        this.indicator = document.getElementById('pageIndicator');
        this.setupScrollListener();
        this.renderPages();
      },

      /**
       * 动态生成页面 (基于pageState.count)
       */
      renderPages() {
        if (!this.viewport) return;
        const currentScrollLeft = this.viewport.scrollLeft;
        const pageWidth = this.viewport.clientWidth;

        // 清空并重新生成页面
        this.viewport.innerHTML = '';
        
        for (let i = 0; i < pageState.count; i++) {
          const page = document.createElement('div');
          page.className = 'desktop-page';
          page.dataset.pageIndex = i;
          
          // 获取该页面所有元素并渲染
          const pageItems = ElementManager.getItemsByPage(i);
          pageItems.forEach(item => {
            const itemEl = this.createDesktopElement(item);
            page.appendChild(itemEl);
          });
          
          this.viewport.appendChild(page);
        }

        // 更新页面指示器
        this.renderIndicator();
        
        // 恢复滚动位置 (避免跳变)
        requestAnimationFrame(() => {
          if (this.viewport) {
            this.viewport.scrollLeft = Math.min(currentScrollLeft, (pageState.count - 1) * pageWidth);
          }
        });
      },

      /**
       * 创建桌面元素DOM
       */
      createDesktopElement(item) {
        const div = document.createElement('div');
        div.className = 'desktop-item';
        div.dataset.itemId = item.id;
        div.dataset.type = item.type;
        
        // 应用网格样式
        const gridStyle = ElementManager.getGridStyle(item);
        div.style.gridColumn = gridStyle.gridColumn;
        div.style.gridRow = gridStyle.gridRow;
        
        // 基础显示内容 (仅显示名称首字或占位，不写死图标)
        div.innerHTML = `
          <div style="font-size: 28px; margin-bottom: 2px; opacity:0.9;">${item.type === 'widget' ? '▣' : '◎'}</div>
          <span style="font-size:11px; text-align:center; line-height:1.2;">${item.name || ''}</span>
        `;
        
        return div;
      },

      /**
       * 页面指示器 (小圆点)
       */
      renderIndicator() {
        if (!this.indicator) return;
        this.indicator.innerHTML = '';
        for (let i = 0; i < pageState.count; i++) {
          const dot = document.createElement('span');
          dot.className = 'indicator-dot';
          if (i === pageState.currentPage) {
            dot.classList.add('active');
          }
          dot.dataset.page = i;
          dot.addEventListener('click', () => this.scrollToPage(i));
          this.indicator.appendChild(dot);
        }
      },

      setupScrollListener() {
        if (!this.viewport) return;
        this.viewport.addEventListener('scroll', () => {
          const pageWidth = this.viewport.clientWidth;
          if (pageWidth <= 0) return;
          const currentPage = Math.round(this.viewport.scrollLeft / pageWidth);
          if (currentPage !== pageState.currentPage && currentPage >= 0 && currentPage < pageState.count) {
            pageState.currentPage = currentPage;
            this.renderIndicator();
          }
        }, { passive: true });
      },

      scrollToPage(index) {
        if (!this.viewport || index < 0 || index >= pageState.count) return;
        const pageWidth = this.viewport.clientWidth;
        this.viewport.scrollTo({ left: index * pageWidth, behavior: 'smooth' });
      },

      /**
       * 动态增加页面
       */
      addPage() {
        pageState.count += 1;
        this.renderPages();
      },

      /**
       * 删除空白页面 (未来逻辑)
       */
      removePage(index) {
        if (pageState.count <= 1) return; // 至少保留一页
        // 检查页面是否有元素 (本阶段简化，未来完善)
        const itemsOnPage = ElementManager.getItemsByPage(index);
        if (itemsOnPage.length > 0) {
          console.warn('页面非空，无法删除');
          return;
        }
        // 调整后续页面索引
        desktopItems.forEach((item, id) => {
          if (item.pageIndex > index) {
            item.pageIndex -= 1;
          }
        });
        pageState.count -= 1;
        if (pageState.currentPage >= pageState.count) {
          pageState.currentPage = pageState.count - 1;
        }
        this.renderPages();
      }
    };

    // ---------- 初始化框架 ----------
    function bootstrap() {
      // 初始化Dock (空状态)
      DockSystem.render();
      
      // 初始化页面系统 (动态生成2页，展示分页架构)
      PageManager.init();
      
      // 框架就绪，无任何示例应用/图标/小组件
      console.log('玉界机桌面框架已启动 - 当前页数:', pageState.count);
      
      // 预留全局访问用于调试/扩展 (未来移除)
      window.__JadeWorldFramework = {
        ElementManager,
        GridSystem,
        DockSystem,
        PageManager,
        desktopItems,
        pageState,
        dockState,
      };
    }

    // 启动应用
    bootstrap();
  </script>
